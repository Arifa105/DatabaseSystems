01 · DDL vs DML Overview

SQL is divided into sublanguages. The two most fundamental are DDL (Data Definition Language) and DML (Data Manipulation Language).

DDL — Define Structure

  • CREATE — make tables/databases
  • ALTER — modify structure
  • DROP — delete objects
  • TRUNCATE — empty a table
  • RENAME — rename an object

DML — Manipulate Data

  • INSERT — add rows
  • SELECT — query rows
  • UPDATE — modify rows
  • DELETE — remove rows
  • MERGE — upsert rows
💡 DDL statements auto-commit in most databases. DML statements can be wrapped in transactions and rolled back.
🏗️

02 · CREATE TABLE

Define a new table, its columns, and their data types.

DDL 
-- Create the database first
CREATE DATABASE company_db;
USE company_db;

-- Create an employees table
CREATE TABLE employees (
    emp_id      INT             PRIMARY KEY AUTO_INCREMENT,
    first_name  VARCHAR(50)   NOT NULL,
    last_name   VARCHAR(50)   NOT NULL,
    email       VARCHAR(100)  UNIQUE NOT NULL,
    department  VARCHAR(50),
    salary      DECIMAL(10,2) DEFAULT 30000.00,
    hire_date   DATE,
    is_active   BOOLEAN         DEFAULT TRUE
);

-- Create a departments table
CREATE TABLE departments (
    dept_id   INT            PRIMARY KEY AUTO_INCREMENT,
    dept_name VARCHAR(80)  NOT NULL UNIQUE,
    location  VARCHAR(100)
);

03 · Constraints

Constraints enforce rules at the database level, ensuring data integrity.

CONSTRAINTS 
CREATE TABLE orders (
    order_id    INT            PRIMARY KEY AUTO_INCREMENT,
    emp_id      INT            NOT NULL,
    FOREIGN KEY (emp_id) REFERENCES employees(emp_id)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    product     VARCHAR(100) NOT NULL,
    quantity    INT            CHECK (quantity > 0),
    price       DECIMAL(10,2) NOT NULL,
    order_date  DATE           DEFAULT (CURRENT_DATE),
    status      ENUM('pending','shipped','delivered') DEFAULT 'pending'
);

-- Named constraint example
ALTER TABLE employees
ADD CONSTRAINT chk_salary
CHECK (salary >= 0);
ConstraintPurpose
PRIMARY KEYUniquely identifies each row; not null
FOREIGN KEYLinks to a primary key in another table
UNIQUEAll values in column must be distinct
NOT NULLColumn cannot store NULL
DEFAULTProvides a default value when none given
CHECKValidates a boolean condition

04 · ALTER TABLE

Modify an existing table structure — add, drop, or change columns.

DDL · ALTER 
-- Add a new column
ALTER TABLE employees
ADD COLUMN phone VARCHAR(20);

-- Add multiple columns at once
ALTER TABLE employees
ADD COLUMN manager_id INT,
ADD COLUMN bonus      DECIMAL(8,2) DEFAULT 0;

-- Change a column's data type
ALTER TABLE employees
MODIFY COLUMN phone VARCHAR(30);

-- Rename a column (MySQL 8+ / PostgreSQL)
ALTER TABLE employees
RENAME COLUMN phone TO mobile_number;

-- Drop a column
ALTER TABLE employees
DROP COLUMN mobile_number;

-- Add a foreign key after creation
ALTER TABLE employees
ADD CONSTRAINT fk_dept
FOREIGN KEY (department)
REFERENCES departments(dept_name);

-- Rename the table itself
ALTER TABLE employees
RENAME TO staff;

05 · DROP & TRUNCATE

DDL · DROP 
-- Drop a table (removes structure + all data)
DROP TABLE orders;

-- Safe drop — no error if table doesn't exist
DROP TABLE IF EXISTS orders;

-- Drop the entire database
DROP DATABASE IF EXISTS company_db;

-- TRUNCATE — removes ALL rows but keeps structure
-- Faster than DELETE; resets AUTO_INCREMENT
TRUNCATE TABLE employees;

DROP TABLE

  • Deletes table structure
  • Deletes all data
  • Cannot be rolled back
  • Removes indexes & triggers

TRUNCATE TABLE

  • Keeps table structure
  • Removes all rows
  • Resets AUTO_INCREMENT
  • Faster than DELETE *

06 · INSERT (DML)

Add one or more rows to a table.

DML · INSERT 
-- Insert a single row (named columns)
INSERT INTO employees (first_name, last_name, email, department, salary, hire_date)
VALUES ('Alice', 'Johnson', 'alice@corp.com', 'Engineering', 85000, '2022-03-01');

-- Insert multiple rows at once
INSERT INTO employees (first_name, last_name, email, department, salary, hire_date)
VALUES
    ('Bob',   'Smith',  'bob@corp.com',   'Marketing',   72000, '2021-07-15'),
    ('Carol', 'Davis',  'carol@corp.com', 'Engineering', 91000, '2020-01-10'),
    ('Dan',   'Wilson', 'dan@corp.com',   'HR',          65000, '2023-05-20'),
    ('Eva',   'Brown',  'eva@corp.com',   'Marketing',   77000, '2019-11-03');

-- INSERT with SELECT — copy rows from another table
INSERT INTO archive_employees
SELECT * FROM employees
WHERE is_active = FALSE;

-- INSERT IGNORE — skip duplicate key errors
INSERT IGNORE INTO employees (email)
VALUES ('alice@corp.com');

07 · SELECT (DML)

Query data from tables. SELECT is the most-used SQL statement.

DML · SELECT 
-- Select all columns
SELECT * FROM employees;

-- Select specific columns
SELECT first_name, last_name, salary FROM employees;

-- Column aliases with AS
SELECT
    first_name                       AS "First",
    last_name                        AS "Last",
    salary * 12                     AS "Annual Salary",
    CONCAT(first_name, ' ', last_name) AS "Full Name"
FROM employees;

-- DISTINCT — remove duplicate values
SELECT DISTINCT department FROM employees;

-- ORDER BY — sort results
SELECT first_name, salary
FROM   employees
ORDER BY salary DESC, first_name ASC;

-- LIMIT — get top N rows
SELECT first_name, salary
FROM   employees
ORDER BY salary DESC
LIMIT 3;

-- LIMIT + OFFSET — pagination (page 3, rows 21-30)
SELECT * FROM employees
LIMIT 10 OFFSET 20;
⚙️

08 · WHERE & Operators

Filter rows using conditions. WHERE supports a rich set of operators.

DML · WHERE 
-- Comparison operators
SELECT * FROM employees WHERE salary > 75000;
SELECT * FROM employees WHERE department = 'Engineering';
SELECT * FROM employees WHERE salary != 65000;

-- AND / OR / NOT
SELECT * FROM employees
WHERE department = 'Engineering' AND salary > 80000;

SELECT * FROM employees
WHERE department = 'HR' OR department = 'Marketing';

-- IN — match against a list
SELECT * FROM employees
WHERE department IN ('HR', 'Marketing', 'Finance');

-- BETWEEN — inclusive range
SELECT * FROM employees
WHERE salary BETWEEN 60000 AND 90000;

-- LIKE — pattern matching (% = any chars, _ = one char)
SELECT * FROM employees WHERE first_name LIKE 'A%';
SELECT * FROM employees WHERE email      LIKE '%@corp.com';
SELECT * FROM employees WHERE last_name  LIKE 'S___h';

-- IS NULL / IS NOT NULL
SELECT * FROM employees WHERE hire_date  IS NULL;
SELECT * FROM employees WHERE manager_id IS NOT NULL;

-- Subquery in WHERE
SELECT first_name, salary
FROM   employees
WHERE  salary > (SELECT AVG(salary) FROM employees);
🔄

09 · UPDATE (DML)

Modify existing rows. Always use a WHERE clause to avoid updating every row!

DML · UPDATE 
-- Update a single column for one row
UPDATE employees
SET    salary = 90000
WHERE  emp_id = 1;

-- Update multiple columns
UPDATE employees
SET
    department = 'Engineering',
    salary     = salary * 1.10
WHERE  emp_id = 2;

-- Update entire department (5% raise)
UPDATE employees
SET    salary = salary * 1.05
WHERE  department = 'Marketing';

-- UPDATE with CASE (conditional update)
UPDATE employees
SET salary = CASE
    WHEN department = 'Engineering' THEN salary * 1.15
    WHEN department = 'Marketing'   THEN salary * 1.08
    ELSE                                   salary * 1.03
END;

-- UPDATE using a subquery
UPDATE employees
SET    is_active = FALSE
WHERE  emp_id IN (SELECT emp_id FROM resigned);
⚠️ Danger: An UPDATE without WHERE updates every row. Always preview with a SELECT first.

10 · DELETE (DML)

Remove rows from a table. Unlike TRUNCATE, DELETE can be filtered and rolled back inside a transaction.

DML · DELETE 
-- Delete a specific row
DELETE FROM employees
WHERE emp_id = 5;

-- Delete rows matching a condition
DELETE FROM employees
WHERE is_active = FALSE;

-- Delete with complex WHERE
DELETE FROM orders
WHERE order_date < '2020-01-01'
  AND status = 'delivered';

-- Delete using a subquery
DELETE FROM employees
WHERE department = (
    SELECT dept_name FROM departments
    WHERE location = 'Closed'
);

-- Safe pattern: SELECT first, DELETE second
SELECT * FROM employees WHERE hire_date < '2019-01-01';
DELETE FROM employees WHERE hire_date < '2019-01-01';

-- Wrap in transaction for safety
START TRANSACTION;
DELETE FROM employees WHERE emp_id = 3;
ROLLBACK;   -- undo if something looks wrong
-- COMMIT;   -- or finalize the deletion

11 · JOINs

Combine rows from two or more tables based on a related column.

DML · JOIN 
-- INNER JOIN — only matching rows in both tables
SELECT e.first_name, e.last_name, o.product, o.quantity
FROM       employees e
INNER JOIN orders    o ON e.emp_id = o.emp_id;

-- LEFT JOIN — all employees, even those with no orders
SELECT e.first_name, o.product
FROM      employees e
LEFT JOIN orders    o ON e.emp_id = o.emp_id;

-- RIGHT JOIN — all orders, even for non-existing employees
SELECT e.first_name, o.product
FROM       employees e
RIGHT JOIN orders    o ON e.emp_id = o.emp_id;

-- FULL OUTER JOIN via UNION (MySQL workaround)
SELECT e.first_name, o.product
FROM      employees e
LEFT JOIN orders    o ON e.emp_id = o.emp_id
UNION
SELECT e.first_name, o.product
FROM       employees e
RIGHT JOIN orders    o ON e.emp_id = o.emp_id;

-- SELF JOIN — manager/employee hierarchy
SELECT
    e.first_name AS "Employee",
    m.first_name AS "Manager"
FROM      employees e
LEFT JOIN employees m ON e.manager_id = m.emp_id;

-- 3-way JOIN
SELECT e.first_name, d.dept_name, o.product
FROM       employees   e
JOIN       departments d ON e.department = d.dept_name
LEFT JOIN  orders      o ON e.emp_id    = o.emp_id;
JOIN TypeReturns
INNER JOINOnly rows with matches in both tables
LEFT JOINAll left rows + matched right rows (NULL if no match)
RIGHT JOINAll right rows + matched left rows (NULL if no match)
FULL OUTER JOINAll rows from both tables
CROSS JOINCartesian product — every combination of rows
SELF JOINTable joined to itself (hierarchy / relationships)

12 · Aggregate Functions & GROUP BY

Perform calculations across groups of rows.

DML · AGGREGATE 
-- Basic aggregate functions
SELECT
    COUNT(*)           AS "Total Employees",
    COUNT(hire_date)  AS "With Hire Date",  -- ignores NULLs
    SUM(salary)       AS "Total Payroll",
    AVG(salary)       AS "Average Salary",
    MIN(salary)       AS "Lowest Salary",
    MAX(salary)       AS "Highest Salary"
FROM employees;

-- GROUP BY — aggregate per department
SELECT
    department,
    COUNT(*)    AS "Head Count",
    AVG(salary) AS "Avg Salary",
    MAX(salary) AS "Top Salary"
FROM     employees
GROUP BY department
ORDER BY AVG(salary) DESC;

-- HAVING — filter groups (like WHERE but for aggregates)
SELECT
    department,
    COUNT(*)    AS "Head Count",
    AVG(salary) AS "Avg Salary"
FROM     employees
GROUP BY department
HAVING   COUNT(*) >= 2
     AND AVG(salary) > 70000;

-- Full SELECT clause execution order:
SELECT   department, COUNT(*)
FROM     employees
WHERE    is_active = TRUE        -- 1. filter rows
GROUP BY department              -- 2. group
HAVING   COUNT(*) > 1           -- 3. filter groups
ORDER BY COUNT(*) DESC          -- 4. sort
LIMIT    5;                      -- 5. limit output
✅ Clause execution order: FROM → JOIN → WHERE → GROUP BY → HAVING → SELECT → ORDER BY → LIMIT

Useful String & Date Functions

FUNCTIONS 
-- String functions
SELECT
    UPPER(first_name),
    LOWER(email),
    LENGTH(last_name),
    TRIM(first_name),
    SUBSTRING(email, 1, 5),
    REPLACE(email, '@corp.com', '')
FROM employees;

-- Date functions
SELECT
    NOW()                          AS "Current DateTime",
    CURDATE()                      AS "Today",
    YEAR(hire_date)                AS "Year Hired",
    MONTH(hire_date)               AS "Month",
    DATEDIFF(CURDATE(), hire_date)  AS "Days Employed"
FROM employees;
