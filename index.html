DDL — Data Definition Language

Commands that define and modify the structure of database objects.

CREATE ALTER DROP TRUNCATE RENAME

CREATE DATABASE & TABLE

Used to create a new database or table. Columns are defined with data types and optional constraints.

CREATE 
-- 1. Create and select a database
CREATE DATABASE school_db;
USE school_db;

-- 2. Students table
CREATE TABLE students (
    student_id  INT             PRIMARY KEY AUTO_INCREMENT,
    name        VARCHAR(100)  NOT NULL,
    email       VARCHAR(150)  UNIQUE NOT NULL,
    dob         DATE,
    grade       CHAR(1)        CHECK (grade IN ('A','B','C','D','F')),
    fee_paid    DECIMAL(8,2)   DEFAULT 0.00,
    enrolled_on TIMESTAMP      DEFAULT CURRENT_TIMESTAMP
);

-- 3. Courses table
CREATE TABLE courses (
    course_id   INT            PRIMARY KEY AUTO_INCREMENT,
    title       VARCHAR(120) NOT NULL,
    credits     INT            DEFAULT 3
);

-- 4. Enrollments junction table (FK on both sides)
CREATE TABLE enrollments (
    id          INT  PRIMARY KEY AUTO_INCREMENT,
    student_id  INT  NOT NULL,
    course_id   INT  NOT NULL,
    enrolled_on DATE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    FOREIGN KEY (course_id)  REFERENCES courses(course_id)  ON DELETE CASCADE
);

-- 5. CREATE TABLE AS SELECT (copy structure + data)
CREATE TABLE students_backup
AS SELECT * FROM students;

-- 6. CREATE TABLE IF NOT EXISTS (safe creation)
CREATE TABLE IF NOT EXISTS logs (
    log_id   INT PRIMARY KEY AUTO_INCREMENT,
    message  TEXT,
    logged   TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 7. CREATE VIEW (virtual table)
CREATE VIEW student_summary AS
SELECT s.name, COUNT(e.course_id) AS total_courses
FROM students s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY s.name;

-- 8. CREATE INDEX (speed up searches)
CREATE INDEX idx_student_email
ON students(email);

ALTER TABLE

Modify an existing table — add/drop/rename columns, change types, add constraints.

ALTER 
-- Add a column
ALTER TABLE students
ADD COLUMN phone VARCHAR(20);

-- Add multiple columns
ALTER TABLE students
ADD COLUMN address  TEXT,
ADD COLUMN city     VARCHAR(80),
ADD COLUMN is_active BOOLEAN DEFAULT TRUE;

-- Modify a column's type
ALTER TABLE students
MODIFY COLUMN phone VARCHAR(30);

-- Rename a column (MySQL 8+)
ALTER TABLE students
RENAME COLUMN dob TO date_of_birth;

-- Drop a column
ALTER TABLE students
DROP COLUMN address;

-- Add a named constraint
ALTER TABLE students
ADD CONSTRAINT chk_fee CHECK (fee_paid >= 0);

-- Drop a constraint
ALTER TABLE students
DROP CONSTRAINT chk_fee;

-- Rename the table
RENAME TABLE students_backup TO students_archive;

DROP & TRUNCATE

DROP / TRUNCATE 
-- Drop a table permanently
DROP TABLE logs;
DROP TABLE IF EXISTS logs;    -- safe version

-- Drop a view
DROP VIEW IF EXISTS student_summary;

-- Drop an index
DROP INDEX idx_student_email ON students;

-- Drop the entire database
DROP DATABASE IF EXISTS old_db;

-- TRUNCATE: delete all rows, keep structure, reset AUTO_INCREMENT
TRUNCATE TABLE enrollments;

DROP vs TRUNCATE vs DELETE

  • DROP removes table structure + data
  • TRUNCATE clears data, keeps structure
  • DELETE removes rows with WHERE filter
  • TRUNCATE resets AUTO_INCREMENT; DELETE does not
  • DDL auto-commits; DELETE can rollback

Constraints Summary

  • PRIMARY KEY — unique, not null identifier
  • FOREIGN KEY — referential integrity
  • UNIQUE — no duplicate values
  • NOT NULL — value required
  • CHECK — validates a condition
  • DEFAULT — fallback value

DML — Data Manipulation Language

Commands that add, change, and remove the actual data inside tables.

INSERT UPDATE DELETE MERGE

INSERT

Add new rows to a table — single row, bulk insert, or from another query.

INSERT 
-- Single row
INSERT INTO students (name, email, grade, fee_paid)
VALUES ('Alice Raza', 'alice@uni.pk', 'A', 15000);

-- Multiple rows at once
INSERT INTO students (name, email, grade, fee_paid)
VALUES
    ('Bob Khan',    'bob@uni.pk',    'B', 12000),
    ('Carol Ahmed', 'carol@uni.pk',  'A', 15000),
    ('Dan Malik',   'dan@uni.pk',    'C', 9000),
    ('Eva Shah',    'eva@uni.pk',    'B', 12000);

-- INSERT ... SELECT (copy from another table)
INSERT INTO students_archive
SELECT * FROM students
WHERE  is_active = FALSE;

-- INSERT IGNORE (skip duplicate key errors)
INSERT IGNORE INTO students (email)
VALUES ('alice@uni.pk');

-- INSERT ... ON DUPLICATE KEY UPDATE (upsert)
INSERT INTO students (email, name, fee_paid)
VALUES ('alice@uni.pk', 'Alice Raza', 16000)
ON DUPLICATE KEY UPDATE fee_paid = 16000;

UPDATE

Modify existing rows. Use WHERE carefully — omitting it updates every row.

UPDATE 
-- Update one column
UPDATE students
SET   grade = 'A'
WHERE student_id = 3;

-- Update multiple columns
UPDATE students
SET   grade    = 'B',
      fee_paid = fee_paid + 2000
WHERE student_id = 2;

-- Update with CASE (conditional logic)
UPDATE students
SET   fee_paid = CASE
    WHEN grade = 'A' THEN fee_paid * 0.90   -- 10% discount
    WHEN grade = 'B' THEN fee_paid * 0.95   -- 5% discount
    ELSE                   fee_paid
END;

-- Update using a subquery
UPDATE students
SET   is_active = FALSE
WHERE student_id NOT IN (
    SELECT DISTINCT student_id FROM enrollments
);

DELETE

DELETE 
-- Delete a specific row
DELETE FROM students
WHERE student_id = 5;

-- Delete with condition
DELETE FROM enrollments
WHERE enrolled_on < '2022-01-01';

-- Delete using a subquery
DELETE FROM students
WHERE student_id IN (
    SELECT student_id FROM graduated
    WHERE year < 2020
);

-- Safe pattern — preview first, then delete
SELECT * FROM students WHERE is_active = FALSE;
DELETE FROM students  WHERE is_active = FALSE;
⚠️ Never run UPDATE or DELETE without a WHERE clause in production. Always preview with a SELECT first. Wrap destructive operations in a transaction (TCL) so you can ROLLBACK.

TCL — Transaction Control Language

Manage database transactions to ensure data consistency and integrity.

BEGIN / START TRANSACTION COMMIT ROLLBACK SAVEPOINT

ACID Properties

ACID Explained

  • Atomicity — all or nothing
  • Consistency — always valid state
  • Isolation — concurrent safety
  • Durability — committed = permanent

When to Use Transactions

  • Bank transfers (debit + credit)
  • Order placement (stock - payment)
  • Bulk data migrations
  • Any multi-step DML operation

COMMIT & ROLLBACK

TCL 
-- Basic transaction pattern
START TRANSACTION;               -- or: BEGIN;

INSERT INTO enrollments
    (student_id, course_id, enrolled_on)
VALUES (1, 3, CURDATE());

UPDATE students
SET    fee_paid = fee_paid + 5000
WHERE  student_id = 1;

COMMIT;       -- saves both changes permanently

-- ─────────────────────────────────────

-- Transaction with error handling
START TRANSACTION;

UPDATE accounts SET balance = balance - 5000
WHERE  account_id = 101;

UPDATE accounts SET balance = balance + 5000
WHERE  account_id = 202;

-- If something goes wrong:
ROLLBACK;      -- undoes BOTH updates

-- If everything is fine:
-- COMMIT;

SAVEPOINT

Create intermediate checkpoints inside a transaction — rollback to a specific point without undoing everything.

SAVEPOINT 
START TRANSACTION;

INSERT INTO courses (title, credits)
VALUES ('Database Systems', 4);

SAVEPOINT after_course;   -- checkpoint 1

INSERT INTO enrollments (student_id, course_id)
VALUES (2, LAST_INSERT_ID());

SAVEPOINT after_enroll;   -- checkpoint 2

UPDATE students SET fee_paid = 0
WHERE  student_id = 99;   -- oops — wrong student

ROLLBACK TO SAVEPOINT after_enroll;  -- undo only the last UPDATE

RELEASE SAVEPOINT after_course;      -- free the savepoint

COMMIT;   -- commits: course insert + enrollment insert

-- ─────────────────────────────────────
-- Turn off auto-commit mode
SET autocommit = 0;   -- every statement is now in a transaction
SET autocommit = 1;   -- re-enable auto-commit
💡 Best Practice: Always wrap multi-step INSERT/UPDATE/DELETE operations in a transaction. Use SAVEPOINT for complex batches where partial rollback might be needed.

DQL — Data Query Language

The art of retrieving data — filtering, joining, grouping, and advanced querying.

SELECT WHERE JOIN GROUP BY HAVING Subqueries Window Fn

SELECT — Fundamentals

SELECT 
-- All columns
SELECT * FROM students;

-- Specific columns + alias
SELECT
    student_id                              AS "ID",
    CONCAT(UPPER(LEFT(name,1)),SUBSTR(name,2)) AS "Name",
    fee_paid                                AS "Fee Paid",
    grade
FROM students
ORDER BY grade ASC, fee_paid DESC;

-- DISTINCT values
SELECT DISTINCT grade FROM students;

-- LIMIT + OFFSET (pagination)
SELECT * FROM students
ORDER BY student_id
LIMIT 5 OFFSET 10;  -- page 3 (rows 11-15)

WHERE — Filtering

WHERE 
SELECT * FROM students WHERE grade = 'A';
SELECT * FROM students WHERE fee_paid BETWEEN 10000 AND 15000;
SELECT * FROM students WHERE name LIKE 'A%';
SELECT * FROM students WHERE grade IN ('A','B');
SELECT * FROM students WHERE date_of_birth IS NULL;
SELECT * FROM students
WHERE  grade = 'A' AND fee_paid > 12000;

-- Subquery in WHERE
SELECT name, fee_paid FROM students
WHERE  fee_paid > (SELECT AVG(fee_paid) FROM students);

-- EXISTS
SELECT name FROM students s
WHERE EXISTS (
    SELECT 1 FROM enrollments e
    WHERE e.student_id = s.student_id
);

JOINs

JOIN 
-- INNER JOIN
SELECT s.name, c.title, e.enrolled_on
FROM       students    s
INNER JOIN enrollments e ON s.student_id = e.student_id
INNER JOIN courses     c ON e.course_id  = c.course_id;

-- LEFT JOIN (all students, even without courses)
SELECT s.name, COUNT(e.course_id) AS courses
FROM      students    s
LEFT JOIN enrollments e ON s.student_id = e.student_id
GROUP BY  s.name;

-- SELF JOIN (students mentoring students)
SELECT
    s.name        AS "Student",
    mentor.name  AS "Mentor"
FROM       students s
LEFT JOIN  students mentor ON s.student_id = mentor.student_id + 1;

GROUP BY, HAVING & Aggregates

AGGREGATE 
-- Basic aggregates
SELECT
    COUNT(*)          AS "Total Students",
    AVG(fee_paid)     AS "Avg Fee",
    SUM(fee_paid)     AS "Total Revenue",
    MIN(fee_paid)     AS "Min Fee",
    MAX(fee_paid)     AS "Max Fee"
FROM students;

-- GROUP BY grade
SELECT   grade,
         COUNT(*)       AS "Count",
         AVG(fee_paid)  AS "Avg Fee"
FROM     students
GROUP BY grade
HAVING   COUNT(*) >= 2
ORDER BY grade;

-- Courses per student (only those with 2+ courses)
SELECT   s.name, COUNT(e.course_id) AS total
FROM     students s
JOIN     enrollments e ON s.student_id = e.student_id
GROUP BY s.name
HAVING   total >= 2
ORDER BY total DESC;

Subqueries & CTEs

ADVANCED 
-- Subquery in FROM (derived table)
SELECT name, fee_paid
FROM (
    SELECT name, fee_paid,
           RANK() OVER (ORDER BY fee_paid DESC) AS rnk
    FROM students
) ranked
WHERE rnk <= 3;

-- CTE (Common Table Expression)
WITH top_students AS (
    SELECT student_id, name, fee_paid
    FROM   students
    WHERE  grade = 'A'
)
SELECT t.name, COUNT(e.course_id) AS courses
FROM      top_students  t
LEFT JOIN enrollments  e ON t.student_id = e.student_id
GROUP BY  t.name;

-- Window Functions
SELECT
    name,
    fee_paid,
    RANK()        OVER (ORDER BY fee_paid DESC)   AS overall_rank,
    RANK()        OVER (PARTITION BY grade ORDER BY fee_paid DESC) AS grade_rank,
    ROW_NUMBER()  OVER (ORDER BY student_id)       AS row_num,
    LAG(fee_paid)  OVER (ORDER BY student_id)       AS prev_fee
FROM students;

DCL — Data Control Language

Manage who can access what — grant and revoke permissions on database objects.

GRANT REVOKE CREATE USER DROP USER SHOW GRANTS

CREATE USER

CREATE USER 
-- Create a user with password
CREATE USER 'teacher'@'localhost'
IDENTIFIED BY 'SecurePass123!';

-- Create a user accessible from any host
CREATE USER 'student_app'@'%'
IDENTIFIED BY 'AppPass456!';

-- Create user only if not exists
CREATE USER IF NOT EXISTS 'readonly_user'@'localhost'
IDENTIFIED BY 'ReadOnly789!';

-- Change a user's password
ALTER USER 'teacher'@'localhost'
IDENTIFIED BY 'NewPass999!';

GRANT — Give Permissions

GRANT 
-- Grant all privileges on a database
GRANT ALL PRIVILEGES ON school_db.*
TO 'teacher'@'localhost';

-- Grant SELECT only (read-only user)
GRANT SELECT ON school_db.*
TO 'readonly_user'@'localhost';

-- Grant SELECT on a specific table
GRANT SELECT ON school_db.students
TO 'student_app'@'%';

-- Grant specific DML privileges
GRANT SELECT, INSERT, UPDATE ON school_db.enrollments
TO 'student_app'@'%';

-- Grant with GRANT OPTION (user can grant to others)
GRANT SELECT, INSERT ON school_db.*
TO 'teacher'@'localhost'
WITH GRANT OPTION;

-- Apply changes immediately
FLUSH PRIVILEGES;

REVOKE — Remove Permissions

REVOKE 
-- Revoke a specific privilege
REVOKE INSERT ON school_db.students
FROM 'student_app'@'%';

-- Revoke all privileges on a database
REVOKE ALL PRIVILEGES ON school_db.*
FROM 'readonly_user'@'localhost';

-- Revoke the GRANT OPTION
REVOKE GRANT OPTION ON school_db.*
FROM 'teacher'@'localhost';

-- Check what a user can do
SHOW GRANTS FOR 'teacher'@'localhost';

-- Drop a user entirely
DROP USER 'readonly_user'@'localhost';
PrivilegeAllows
SELECTRead data from tables
INSERTAdd new rows
UPDATEModify existing rows
DELETERemove rows
CREATECreate tables, databases, views
DROPDelete tables or databases
ALTERModify table structure
INDEXCreate and drop indexes
EXECUTERun stored procedures/functions
ALL PRIVILEGESEvery privilege listed above
🛡️ Principle of Least Privilege: Only grant the minimum permissions required. A reporting user needs SELECT only. A web app insert-only. Never give ALL PRIVILEGES to application users.
